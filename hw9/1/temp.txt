class Picture:
    
    def __init__(self):
        '''
        __init__(self)
        '''
        self.shapes = []

    def get_shapes(self):
        '''
        get_shapes(self)
            Get Shapes included in the Picture.

        returns:
            A list of Shape object.
        '''
        return self.shapes

    def add_shape(self, shape):
        '''
        add_shape(self, shape):
            Add a Shape to shape-list.

        args:
            shape: Shape
                Shape object to add.

        returns:
            self.
        '''
        self.shapes.append(shape)
        return self

    def rotate(self, theta, unit='deg'):
        '''
        rotate(self, theta, unit='deg')
            Rotate all shapes in the picture. The Picture will not
            be changed.

        args:
            theta: float
                Angle to rotate.

            unit: str from ['deg', 'rad']
                theta`s unit.

        returns:
            A Picture object.
        '''
